	usecase MesToTgBot as "Enviar un mensaje a la cuenta bot de Telegram"
	usecase SaveMesTg as "Almacenar en \nlos servidores de Telegram"
	Usuario -> MesToTgBot
	MesToTgBot ..> SaveMesTg : include
}

	usecase SaveMesTw as "Almacenar en \nlos servidores de Twitter"
	Usuario --> MesToTwBot
	MesToTwBot ..> SaveMesTw : include
}
